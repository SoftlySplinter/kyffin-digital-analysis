00115                 return Threading::eEventSuccess;

00135 
00136                         // mutex can be unlocked
00137                         pthread_mutex_unlock(&m_PosixMutex);
00138                         
00139                         // return with success
00140                         return Threading::eEventSuccess;
00141                 }
00142                 else                    // wait with timeout
00143                 {
00144                         // set timer structure
00145                         time_t current = time(NULL);
00146                         timespec uni_ts;

00161

00163                         {
