00776 void CLinux1394Capture2::YUVToRGB(int y, int u, int v, unsigned char* output)
00777 {
00778         int r,g,b;
00779         
00780         // u and v are +-0.5
00781         u -= 128;
00782         v -= 128;
00783         
00784         // conversion
00785         r = int(y + 1.370705 * v);
00786         g = int(y - 0.698001 * v - 0.337633 * u);
00787         b = int(y + 1.732446 * u);
00788         
00789         // clamp to 0..1
00790         if (r < 0) r = 0;
00791         if (g < 0) g = 0;
00792         if (b < 0) b = 0;
00793         if (r > 255) r = 255;
00794         if (g > 255) g = 255;
00795         if (b > 255) b = 255;
00796         
00797         output[0] = (unsigned char) r;
00798         output[1] = (unsigned char) g;
00799         output[2] = (unsigned char) b;
00800 }
