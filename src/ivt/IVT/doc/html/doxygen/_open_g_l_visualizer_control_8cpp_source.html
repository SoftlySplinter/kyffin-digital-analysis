00204         
00205         if (m_mouse_mode == 1)
00206         {
00207                 Mat3d m1, m2;

00209                 Math3d::SetRotationMatX(m1, ROT_SPEED * fy);
00210                 Math3d::SetRotationMatY(m2, ROT_SPEED * -fx);

00212                 Math3d::MulMatMat(m1, m2, m1);

00217         }
00218         else if (m_mouse_mode == 2)
