05123         }
05124 
05125         const int nKernelSize = 5;
05126         const int k = 2;//(nKernelSize - 1) / 2;

05128         float pFilter[nKernelSize];

05130         int i;
05131         
05132         // construct Gaussian kernel
05133         for (i = 0; i < nKernelSize; i++)
05134         {
05135                 pFilter[i] = expf(-(i - k) * (i - k) / (2.0f * fVariance));
05136                 sum += pFilter[i];
05137         }
05138   
05139         for (i = 0; i < nKernelSize; i++) 
05140                 pFilter[i] /= sum;

05142         // create temp image if necessary
05143         CFloatMatrix *pTempImage = new CFloatMatrix(pInputImage);
05144         
05145         const int width = pInputImage->columns;
05146         const int height = pInputImage->rows;
05147 
05148         const float *input = pInputImage->data;

05151         
05152         int x, y, d, offset;
05153         
05154         
05155         // x direction
05156         for (y = 0, offset = 0; y < height; y++)
05157         {
05158                 for (x = 0; x < k; x++, offset++)
05159                 {
05160                         float sum = 0.0f;
05161       
05162                         for (d = -k; d < -x; d++) 
05163                                 sum += input[y * width] * pFilter[k + d];
05164       
05165                         for (d = -x; d <= k; d++) 
05166                                 sum += input[offset + d] * pFilter[k + d];
05167       
05168                         temp[offset] = sum;
05169                 }
05170     
05171                 // core loop (optimized)
05172                 for (x = 2 * k; x < width; x++, offset++)
05173                 {
05174                         temp[offset] =
05175                                 input[offset - 2] * pFilter[0] +
05176                                 input[offset - 1] * pFilter[1] +
05177                                 input[offset] * pFilter[2] +
05178                                 input[offset + 1] * pFilter[3] +
05179                                 input[offset + 2] * pFilter[4];
05180                 }
05181     
05182                 for (x = width - k; x < width; x++, offset++)
05183                 {
05184                         float sum = 0.0f;
05185       
05186                         for (d = -k; d < width - x; d++) 
05187                                 sum += input[offset + d] * pFilter[k + d];
05188       
05189                         for (d = width - x; d <= k; d++) 
05190                                 sum += input[y * width + width - 1] * pFilter[k + d];
05191       
05192                         temp[offset] = sum;
05193                 }
05194         }
05195         
05196   
05197         // y direction
05198         for (y = 0; y < k; y++)
05199         {
05200                 for (x = 0; x < width; x++)
05201                 {
05202                         float sum = 0.0f;
05203       
05204                         for (d = -k; d < -y; d++) 
05205                                 sum += temp[x] * pFilter[k + d];

05207                         for (d = -y; d <= k; d++) 
05208                                 sum += temp[(y + d) * width + x] * pFilter[k + d];
05209         
05210                         output[y * width + x] = sum;
05211                 }
05212         }
05213     
05214         // core loop (optimized)
05215         for (y = 2 * k, offset = k * width; y < height; y++)
05216         {
05217                 for (x = 0; x < width; x++, offset++)

06166 static inline void CalculateSubpixel(const unsigned char *input, int width, int sector, Vec2d &point, Vec2d &direction)
06167 {
06168         int a, b, c, n;
06169         
06170         if (sector == 0)
06171         {
06172                 a = *input - *(input - 2);
06173                 b = *(input + 1) - *(input - 1);
06174                 c = *(input + 2) - *input;
06175                 n = a - 2 * b + c;
06176                 
06177                 if (n != 0)
06178                 {
06179                         const float x = 0.5f * (a - c) / n;
06180 
06181                         if (fabsf(x) < 1.0f)
06182                                 point.x += x;
06183                 }
06184         }
06185         else if (sector == 1)
06186         {
06187                 a = *input - *(input - (width << 1));
06188                 b = *(input + width) - *(input - width);
06189                 c = *(input + (width << 1)) - *input;
06190                 n = a - 2 * b + c;
06191                 
06192                 if (n != 0)
06193                 {
06194                         const float y = 0.5f * (a - c) / n;
06195 
06196                         if (fabsf(y) < 1.0f)
06197                                 point.y += y;
06198                 }
06199         }
06200         else if (sector == 2)
06201         {
06202                 a = *input - *(input - (width << 1) - 2);
06203                 b = *(input + width + 1) - *(input - width - 1);
06204                 c = *(input + (width << 1) + 2) - *input;
06205                 n = a - 2 * b + c;
06206                 
06207                 if (n != 0)
06208                 {
06209                         const float xy = 0.35355339059327378637f * (a - c) / n;
06210 
06211                         if (fabsf(xy) < 1.0f)
06212                         {
06213                                 point.y += xy;
06214                                 point.x += xy;
06215                         }
06216                 }
06217         }
06218         else if (sector == 3)
06219         {
06220                 a = *input - *(input - (width << 1) + 2);
06221                 b = *(input + width - 1) - *(input - width + 1);
06222                 c = *(input + (width << 1) - 2) - *input;
06223                 n = a - 2 * b + c;
06224                 
06225                 if (n != 0)
06226                 {
06227                         const float xy = 0.35355339059327378637f * (a - c) / n;
06228 
06229                         if (fabsf(xy) < 1.0f)
06230                         {
06231                                 point.y += xy;
06232                                 point.x -= xy;
06233                         }
06234                 }
06235         }
06236 
06237         Math2d::SetVec(direction,
06238                 (float) ((input[1] << 1) + input[1 - width] + input[1 + width] - (input[-1] << 1) - input[-1 - width] - input[-1 + width]),
06239                 (float) ((input[width] << 1) + input[width - 1] + input[width + 1] - (input[-width] << 1) - input[-width - 1] - input[-width + 1]));
06240         Math2d::NormalizeVec(direction);
06241 }

06373 void ImageProcessor::CalculateHarrisMap(const CByteImage *pInputImage, CIntImage *pOutputImage)
06374 {
06375         if (pInputImage->type != CByteImage::eGrayScale || pInputImage->width != pOutputImage->width || pInputImage->height != pOutputImage->height)
06376         {
06377                 printf("error: input image and output matrix do not match for ImageProcessor::CalculateHarrisMap\n");
06378                 return;
06379         }
06380         
06381         const int width = pInputImage->width;
06382         const int height = pInputImage->height;
06383         const int nPixels = width * height;
06384         
06385         const unsigned char *input = pInputImage->pixels;
06386 
06387         int i, j, offset;
06388 
06389         int *cov = new int[3 * nPixels];
06390         
06391         // calculate gradients
06392         const int stop = nPixels - width - 1;
06393         for (offset = width + 1; offset < stop; offset++)
06394         {
06395                 const int p = int(input[offset + 1]) - int(input[offset - 1]);
06396                 const int q = int(input[offset + width]) - int(input[offset - width]);
06397                 const int offset_ = 3 * offset;
06398 
06399                 cov[offset_    ] = p * p;
06400                 cov[offset_ + 1] = q * q;
06401                 cov[offset_ + 2] = p * q;
06402         }
06403 
06404         // zero R image
06405         Zero(pOutputImage);
06406 
06407         // calculate harris response function
06408         const int diff = HARRIS_WINDOW_SIZE + 1;
06409         const int width3 = 3 * width;
06410         int *R = pOutputImage->pixels + (HARRIS_WINDOW_SIZE / 2) * (width + 1);
06411 
06412         // do not use one-pixel-thick border of cov (gradients are not valid there)
06413         for (i = diff, offset = width + 1; i < height; i++, offset += diff)
06414         {
06415                 for (j = diff; j < width; j++, offset++)
06416                 {
06417                         int ppsum = 0;
06418                         int qqsum = 0;
06419                         int pqsum = 0;
06420 
06421                         for (int k = 0, offset2 = 3 * offset; k < HARRIS_WINDOW_SIZE; k++, offset2 += width3)
06422                         {
06423                                 ppsum += cov[offset2];
06424                                 qqsum += cov[offset2 + 1];
06425                                 pqsum += cov[offset2 + 2];
06426                                 ppsum += cov[offset2 + 3];
06427                                 qqsum += cov[offset2 + 4];
06428                                 pqsum += cov[offset2 + 5];
06429                                 ppsum += cov[offset2 + 6];
06430                                 qqsum += cov[offset2 + 7];
06431                                 pqsum += cov[offset2 + 8];
06432                         }
06433                         
06434                         // NOTE:
06435                         // while abs()  should not be necessary for most targets (the squared result fits into 32bit),
06436                         // some targets (known:  ti 64x) produce faulty code here.
06437                         pqsum = abs(pqsum);
06438                         
06439                         ppsum >>= 4;
06440                         qqsum >>= 4;
06441                         pqsum >>= 4;
06442                         
06443                         //R[offset] = (ppsum * qqsum - pqsum * pqsum) - 0.04f * ((ppsum + qqsum) * (ppsum + qqsum));
06444                         // approximation: 0.04f = 1 / 25. instead: 1 / 16
06445                         R[offset] = (ppsum * qqsum - pqsum * pqsum) - (((ppsum + qqsum) >> 2) * ((ppsum + qqsum) >> 2));
06446                 }
06447         }
06448 
06449         delete [] cov;
06450 }

06453 static void QuicksortInverse(int *pOffsets, const int *pValues, int nLow, int nHigh)
06454 {
06455         int i = nLow;
06456         int j = nHigh;
06457         
06458         const int x = pValues[pOffsets[(nLow + nHigh) >> 1]];
06459 
06460         while (i <= j)
06461         {    
06462                 while (pValues[pOffsets[i]] > x) i++; 
06463                 while (pValues[pOffsets[j]] < x) j--;
06464         
06465                 if (i <= j)
06466                 {
06467                         const int temp = pOffsets[i];
06468                         pOffsets[i] = pOffsets[j];
06469                         pOffsets[j] = temp;
06470                 
06471                         i++; 
06472                         j--;
06473                 }
06474         }
06475 
06476         if (nLow < j) QuicksortInverse(pOffsets, pValues, nLow, j);
06477         if (i < nHigh) QuicksortInverse(pOffsets, pValues, i, nHigh);
06478 }

07086 void ImageProcessor::ConvertBayerPattern(const CByteImage *pBayerImage, CByteImage *pOutputImage, BayerPatternType type)
07087 {
07088         if (pBayerImage->type != CByteImage::eGrayScale || pOutputImage->type != CByteImage::eRGB24)
07089         {
07090                 printf("error: input image must be of type eGrayScale and output image of type eRGB24 for ImageProcessor::ConvertBayerPattern\n");
07091                 return;
07092         }
07093         
07094         const unsigned char *input = pBayerImage->pixels;
07095         unsigned char *output = pOutputImage->pixels;
07096         
07097         const int width = pBayerImage->width;
07098         const int height = pBayerImage->height;
07099         const int width3 = 3 * width;
07100         
07101         int blue = type == eBayerGB || type == eBayerBG ? 1 : -1;
07102         bool bGreenFirst = type == eBayerGR || type == eBayerGB;
07103         
07104         memset(output, 0, width);
07105         memset(output + (height - 1) * width, 0, width);
07106         
07107         for (int i = 2, input_offset = 0, output_offset = width3 + 4; i < height; i++, input_offset += width, output_offset += width3)
07108         {
07109                 const unsigned char *bayer = input + input_offset;
07110                 const unsigned char *bayer_end = bayer + width - 4;
07111                 unsigned char *dst = output + output_offset;
07112                 
07113                 dst[-4] = dst[-3] = dst[-2] = 0;
07114                 
07115                 if (bGreenFirst)
07116                 {
07117                         dst[blue] = (unsigned char) ((bayer[1] + bayer[2 * width + 1] + 1) >> 1);
07118                         dst[0] = bayer[width + 1];
07119                         dst[-blue] = (unsigned char) ((bayer[width] + bayer[width + 2] + 1) >> 1);
07120                         
07121                         bayer++;
07122                         dst += 3;
07123                 }
07124                 
07125                 if (blue > 0)
07126                 {
07127                         for (; bayer <= bayer_end; bayer += 2, dst += 6)
07128                         {
07129                                 register int v = bayer[2] + bayer[2 * width + 2] + 1;
07130                                 
07131                                 dst[4] = (unsigned char) (v >> 1);
07132                                 dst[3] = bayer[width + 2];
07133                                 dst[2] = (unsigned char) ((bayer[width + 1] + bayer[width + 3] + 1) >> 1);
07134
