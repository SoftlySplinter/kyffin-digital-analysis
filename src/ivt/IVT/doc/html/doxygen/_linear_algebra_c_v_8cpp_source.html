00132 void LinearAlgebraCV::CalculatePseudoInverseSVD(const CFloatMatrix *pInputMatrix, CFloatMatrix *pResultMatrix)
00133 {
00134         if (pInputMatrix->columns != pResultMatrix->rows || pInputMatrix->rows != pResultMatrix->columns)
00135         {
00136                 printf("error: input and output matrix do not match LinearAlgebraCV::CalculatePseudoInverseSVD");
00137                 return;
00138         }
00139         
00140         // algorithm:
00141         // 1: compute U * W * V^T = A
00142         // 2: compute W' = W^T with all non-zero values inverted
00143         // 3: compute V * W' * U^T (=pseudoinverse of A)
00144         
00145         const CFloatMatrix *A = pInputMatrix;
00146         const int m = pInputMatrix->rows;
00147         const int n = pInputMatrix->columns;
00148         
00149         CFloatMatrix W(n, m), WT(m, n), UT(m, m), V(n, n);
00150         
00151         // calculate SVD
00152         SVD(A, &W, &UT, &V, false, true, false);
00153         
00154         // transpose middle diagonal matrix
00155         Transpose(&W, &WT);
00156         
00157         const int min = MY_MIN(m, n);
00158 
00159         int i;
00160         float fThreshold = 0.0f;
00161 
00162         for (i = 0; i < min; i++)
00163                 fThreshold += WT(i, i);
00164         
00165         fThreshold *= 2 * FLT_EPSILON;
00166         
00167         // invert non-zero values (along diagonal)
00168         for (i = 0; i < min; i++)
00169                 if (WT(i, i) < fThreshold)
00170                         WT(i, i) = 0.0f;
00171                 else
00172                         WT(i, i) = 1.0f / WT(i, i);
00173 
00174         // calculate pseudoinverse
00175         CFloatMatrix temp(m, n);
00176         Multiply(&V, &WT, &temp);
00177         Multiply(&temp, &UT, pResultMatrix);
00178 }
