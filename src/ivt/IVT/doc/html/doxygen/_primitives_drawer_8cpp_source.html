00611 {

00624 // ****************************************************************************

00626 // can be created by connecting the vertices in pPoints in their respective

00709 // horizontally between each pair of intersection points are filled with the

00820 // ****************************************************************************
00821 // Implementation of the scan line fill algorithm by Derick Beng Yuh.
00822 //
00823 // pImage: the target image
00824 // pPoints: array of points of the polygon with order of connection
00825 // nPoints: the number of points of the polygon
00826 // r, g, b: target color in RGB
00827 // ****************************************************************************
00828 static void FillPolygon(CByteImage *pImage, const Vec2d *pPoints, int nPoints, int r, int g, int b) 
00829 {
00830         const int height = pImage->height;
00831 
00832         int i;
00833         
00834         // allocate memory for edges
00835         Edge **ppEdges = new Edge*[height];
00836         for (i = 0; i < height; i++)
00837                 ppEdges[i] = new Edge();
00838 
00839         // convert float points to int
00840         Vec2dInt *pPointsInt = new Vec2dInt[nPoints];
00841         for (i = 0; i < nPoints; i++)
00842         {
00843                 pPointsInt[i].x = int(pPoints[i].x + 0.5f);
00844                 pPointsInt[i].y = int(pPoints[i].y + 0.5f);
00845         }
00846 
00847         if (pPoints[0].x == pPoints[nPoints - 1].x && pPoints[0].y == pPoints[nPoints - 1].y)
00848                 nPoints--;
00849    
00850         // constuct edges 
00851         BuildEdgeList(pPointsInt, nPoints, ppEdges, height);
00852 
00853         // free memory
00854         delete [] pPointsInt;
00855 
00856         Edge *pActiveEdge = new Edge();
00857 
00858         // run algorithm
00859         for (i = 0; i < height; i++) 
00860         {
00861                 BuildActiveList(i, pActiveEdge, ppEdges);
00862        
00863                 if (pActiveEdge->next) 
00864                 {
00865                         if (pImage->type == CByteImage::eGrayScale)
00866                                 FillScan(pImage, i, pActiveEdge, (r + (g << 1) + b + 2) >> 2);
00867                         else
00868                                 FillScan(pImage, i, pActiveEdge, r, g, b);
00869 
00870                         UpdateActiveList(i, pActiveEdge);
00871                         ResortActiveList(pActiveEdge);
00872                 }
00873         }
00874    
00875         // free memory for edges
00876         for (i = 0; i < height; i++)
00877                 delete ppEdges[i];
00878         delete [] ppEdges;
00879         delete pActiveEdge;
00880 }
