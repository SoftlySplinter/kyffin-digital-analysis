00116                 cvFillConvexPoly(pIplImage, points, nPoints, CV_RGB(b, g, r));
00117         else
00118         {
00119                 for (int i = 0; i < nPoints - 1; i++)
00120                         cvLine(pIplImage, points[i], points[i + 1], CV_RGB(b, g, r), thickness);
00121                         
00122                 if (points[0].x != points[nPoints - 1].x || points[0].y != points[nPoints - 1].y)
00123                         cvLine(pIplImage, points[nPoints - 1], points[0], CV_RGB(b, g, r), thickness);
00124         }
00125         cvReleaseImageHeader(&pIplImage);
