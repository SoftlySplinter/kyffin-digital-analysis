00349 {
00350         float R, G, B;
00351         float S = s / 255.0f, V = v / 255.0f;
00352         
00353         if (h < 120)
00354         {
00355                 R = (120 - h) / 60.0f;
00356                 G = h / 60.0f;
00357                 B = 0;
00358         }
00359         else if (h < 240)
00360         {
00361                 R = 0;
00362                 G = (240 - h) / 60.0f;
00363                 B = (h - 120) / 60.0f;
00364         }
00365         else
00366         {
00367                 R = (h - 240) / 60.0f;
00368                 G = 0.0f;
00369                 B = (360 - h) / 60.0f;
00370         }
00371         
00372         if (R > 1) R = 1.0f;
00373         if (G > 1) G = 1.0f;
00374         if (B > 1) B = 1.0f;

00376         r = int(255.0f * (1.0f + S * (R - 1.0f)) * V + 0.5f);
00377         g = int(255.0f * (1.0f + S * (G - 1.0f)) * V + 0.5f);
00378         b = int(255.0f * (1.0f + S * (B - 1.0f)) * V + 0.5f);
